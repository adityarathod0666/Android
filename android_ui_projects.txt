public class SimpleCalculator extends AppCompatActivity {
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_simple_calculator);

}

// File 3: TemperatureConverter.java
// Solution for Question 5
public class TemperatureConverter extends AppCompatActivity {
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_temperature_converter);

        EditText celsiusInput = findViewById(R.id.celsius_input);
        Button convertButton = findViewById(R.id.convert_button);
        TextView fahrenheitOutput = findViewById(R.id.fahrenheit_output);

        convertButton.setOnClickListener(v -> {
            double celsius = Double.parseDouble(celsiusInput.getText().toString());
            double fahrenheit = (celsius * 9/5) + 32;
            fahrenheitOutput.setText(String.format("%.2f", fahrenheit));
        });
    }
}

// File 5: ToDoList.java
// Solution for Question 7
public class ToDoList extends AppCompatActivity {
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_to_do_list);

        EditText taskInput = findViewById(R.id.task_input);
        Button addTaskButton = findViewById(R.id.add_task_button);
        ListView taskListView = findViewById(R.id.task_list);

        ArrayList<String> tasks = new ArrayList<>();
        ArrayAdapter<String> adapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, tasks);
        taskListView.setAdapter(adapter);

        addTaskButton.setOnClickListener(v -> {
            String task = taskInput.getText().toString();
            if (!task.isEmpty()) {
                tasks.add(task);
                adapter.notifyDataSetChanged();
                taskInput.setText("");
            }
        });
    }
}

// File 6: DatabaseOperations.java
// Solution for Question 8
public class DatabaseOperations extends AppCompatActivity {
    SQLiteDatabase database;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_database_operations);

        database = openOrCreateDatabase("AppDB", MODE_PRIVATE, null);
        database.execSQL("CREATE TABLE IF NOT EXISTS Users(Id INTEGER PRIMARY KEY, Name TEXT);");

        Button insertButton = findViewById(R.id.insert_button);
        Button displayButton = findViewById(R.id.display_button);
        EditText nameInput = findViewById(R.id.name_input);
        TextView displayOutput = findViewById(R.id.display_output);

        insertButton.setOnClickListener(v -> {
            String name = nameInput.getText().toString();
            database.execSQL("INSERT INTO Users(Name) VALUES('" + name + "');");
            Toast.makeText(this, "Data Inserted", Toast.LENGTH_SHORT).show();
        });

        displayButton.setOnClickListener(v -> {
            Cursor cursor = database.rawQuery("SELECT * FROM Users;", null);
            StringBuilder data = new StringBuilder();
            while (cursor.moveToNext()) {
                data.append(cursor.getInt(0)).append(" ").append(cursor.getString(1)).append("\n");
            }
            displayOutput.setText(data.toString());
            cursor.close();
        });
    }
}

// File 7: QuizCompetition.java
// Solution for Question 9
public class QuizCompetition extends AppCompatActivity {
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_quiz_competition);

        TextView questionText = findViewById(R.id.question_text);
        RadioGroup optionsGroup = findViewById(R.id.options_group);
        Button submitButton = findViewById(R.id.submit_button);

        String[] questions = {"Question 1", "Question 2"};
        String[][] options = {{"Option A", "Option B", "Option C"}, {"Option D", "Option E", "Option F"}};
        int currentQuestion = 0;

        submitButton.setOnClickListener(v -> {
            int selectedId = optionsGroup.getCheckedRadioButtonId();
            if (selectedId != -1) {
                currentQuestion++;
                if (currentQuestion < questions.length) {
                    questionText.setText(questions[currentQuestion]);
                } else {
                    Toast.makeText(this, "Quiz Completed", Toast.LENGTH_SHORT).show();
                }
            }
        });
    }
}
